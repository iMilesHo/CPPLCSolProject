什么是二叉堆？
二叉堆是一种完全二叉树，满足堆属性。根据堆属性，二叉堆可以分为两种类型：

- 最小堆（Min-Heap）：每个节点的值都小于或等于其子节点的值。
- 最大堆（Max-Heap）：每个节点的值都大于或等于其子节点的值。

```cpp
// C++ 实现二叉堆 - 最小堆（从索引1开始）
class MinHeap {
private:
    vector<int> heap;  // 索引0不使用，从索引1开始

    void heapifyUp(int index) {
        if (index <= 1) return;
        int parent = index / 2;  // 父节点索引计算更简单
        if (heap[index] < heap[parent]) {
            swap(heap[index], heap[parent]);
            heapifyUp(parent);
        }
    }

    void heapifyDown(int index) {
        int left = 2 * index;      // 左子节点
        int right = 2 * index + 1; // 右子节点
        int smallest = index;

        if (left < heap.size() && heap[left] < heap[smallest]) {
            smallest = left;
        }
        if (right < heap.size() && heap[right] < heap[smallest]) {
            smallest = right;
        }

        if (smallest != index) {
            swap(heap[index], heap[smallest]);
            heapifyDown(smallest);
        }
    }

public:
    MinHeap() {
        heap.push_back(0); // 占位符，让索引从1开始
    }

    void insert(int val) {
        heap.push_back(val);
        heapifyUp(heap.size() - 1);
    }

    int extractMin() {
        if (heap.size() <= 1) return -1;
        int result = heap[1];
        heap[1] = heap.back();
        heap.pop_back();
        if (heap.size() > 1) {
            heapifyDown(1);
        }
        return result;
    }

    int peek() {
        return heap.size() <= 1 ? -1 : heap[1];
    }

    bool empty() {
        return heap.size() <= 1;
    }

    void printHeap() {
        for (int i = 1; i < heap.size(); i++) {  // 从索引1开始打印
            cout << heap[i] << " ";
        }
        cout << endl;
    }
};
```
